CREATE OR REPLACE FUNCTION pgr_fromAtoB(
    IN tbl varchar,
    IN x1 float, IN y1 float,
    IN x2 float, IN y2 float,
)
RETURNS SETOF record AS $$
BEGIN
    FOR rec IN EXECUTE sql LOOP
        RETURN NEXT;
    END LOOP;
END;
$$ LANGUAGE 'plpgsql' VOLATILE STRICT;
